# uwsgi 配置
[uwsgi]
#配置用户组和用户
uid=ubuntu
gid=ubuntu
#开启主进程
master= true
#配置工作进程数
processes = 4
#指定项目路径
chdir = /home/ubuntu/mywebsite
#配置套接字
sock = 127.0.0.1:8000
#指定uwsgi接口py文件
model = mywebsite.wsgi
#服务器退出后自动删除unix socket文件和pid文件
vacuum = true 
#使服务器后天运行，并将日志文件输出到指定的日志文件
daemonize = /var/log/uwsgi.log
pidfile = /var/run/uwsgi.pid
